local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")

local enabled = false

-- GUI زر
local ScreenGui = Instance.new("ScreenGui", game.CoreGui)
local Button = Instance.new("TextButton", ScreenGui)

Button.Size = UDim2.new(0,150,0,50)
Button.Position = UDim2.new(0,20,0,200)
Button.Text = "OFF"
Button.BackgroundColor3 = Color3.fromRGB(255,0,0)

Button.MouseButton1Click:Connect(function()
    enabled = not enabled
    Button.Text = enabled and "ON" or "OFF"
    Button.BackgroundColor3 = enabled and Color3.fromRGB(0,255,0) or Color3.fromRGB(255,0,0)
end)

-- أقرب لاعب
function getClosestPlayer()
    local closest = nil
    local shortestDistance = math.huge

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            
            local distance = (LocalPlayer.Character.HumanoidRootPart.Position - player.Character.HumanoidRootPart.Position).Magnitude
            
            if distance < shortestDistance then
                shortestDistance = distance
                closest = player
            end
        end
    end
    
    return closest
end

-- تشغيل
RunService.Heartbeat:Connect(function()
    if enabled and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid") then
        
        local humanoid = LocalPlayer.Character.Humanoid
        humanoid.WalkSpeed = 59 -- السرعة
        
        local target = getClosestPlayer()
        
        if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
            
            -- يركض باتجاهه
            humanoid:MoveTo(target.Character.HumanoidRootPart.Position)

            -- إذا قرب منه
            local distance = (LocalPlayer.Character.HumanoidRootPart.Position - target.Character.HumanoidRootPart.Position).Magnitude
            
            if distance < 5 then
                -- الضرب
                game:GetService("ReplicatedStorage"):WaitForChild("Attack"):FireServer()
                
                -- ميدوسا (عدل الاسم لو مختلف)
                game:GetService("ReplicatedStorage"):WaitForChild("Medusa"):FireServer(target)
            end
        end
    end
end)
